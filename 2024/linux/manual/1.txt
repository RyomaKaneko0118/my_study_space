【ルートユーザーをログインできないようにする】
cd /etc/ssh
cp ssh_config backup/sshd_config.bak
sudo vim sshd_config
PasswordAuthentication yes
PermitRootLogin no
systemctl restart sshd

【ubuntuユーザーのロック】
usermod -L ubuntu

【ロックされたユーザーの確認】
/etc/shadow

【鍵の作成】

【簡単な方法でログインできるようにする】
ローカル環境

Host myServer
HostName os3-384-25300.vs.sakura.ne.jp
User bld2022
Port 22
IdentityFile ~/.ssh/id_ed25519_my_key_1
ServerAliveInterval 60
ServerAliveCountMax 120

【fail2ban】
sudo apt install fail2ban
sudo systemctl status fail2ban
sudo systemctl enable fail2ban
cd /etc/fail2ban
sudo cp jail.conf jail.local.bak
ブロックしたユーザーのアクセス不可期間を1週間にする。
不正なアクセスを検出する時間を3時間。
3回を超えて連続したログインの失敗があった場合、一時的にブロックする。
[DEFAULT]
ignorecommand =
bantime  = 604800s
findtime  = 10800s
maxretry = 3
[sshd]
mode = aggressive
port    = ssh
logpath = %(sshd_log)s
backend = %(sshd_backend)s
enabled = true

sudo systemctl restart fail2ban
