helloworldという仮想環境を作成
sudo ip netns add helloworld
ip netns list
sudo ip netns exec helloworld ip address show
sudo ip netns exec helloworld ip route show
sudo ip netns exec helloworld bash
sudo ip netns delete helloworld

ns1、ns2という仮想環境を作る。
ns1-veth0というイーサネットデバイスを作成し、
始点をns1-veth0、終点をns2-veth0とする。
イーサネットデバイスの始点をns1、終点をns2と紐付ける。
ns1側のns1-veth0イーサネットデバイスのエンドポイントを、192.0.2.1/24、
ns2側のns1-veth0イーサネットデバイスのエンドポイントを、192.0.2.2/24、
に設置する。
それぞれのエンドポイントを起動する。

sudo ip netns add ns1
sudo ip netns add ns2
sudo ip link add ns1-veth0 type veth peer name ns2-veth0
ip link show

sudo ip link set ns1-veth0 netns ns1 
sudo ip link set ns2-veth0 netns ns2 

sudo ip netns exec ns1 ip address add 192.0.2.1/24 dev ns1-veth0 
sudo ip netns exec ns2 ip address add 192.0.2.2/24 dev ns2-veth0 

sudo ip netns exec ns1 ip link show ns1-veth0

sudo ip netns exec ns1 ip link set ns1-veth0 up
sudo ip netns exec ns2 ip link set ns2-veth0 up

sudo ip netns exec ns1 ping -c 3 192.0.2.2
sudo ip netns delete ns1
sudo ip netns delete ns2
sudo ip link delete ns1-veth0

ns1、ns2、routerという仮想環境を作る。

ns1-veth0というイーサネットデバイスを作成し、
始点をns1-veth0、終点をgw-veth0とする。
ns1-veth0イーサネットデバイスの始点をns1、終点をrouterと紐付ける。
ns1側のns1-veth0イーサネットデバイスのエンドポイントを、192.0.2.1/24、
router側のns1-veth0イーサネットデバイスのエンドポイントを、192.0.2.254/24、
に設置する。
それぞれのエンドポイントを起動する。

ns2-veth0というイーサネットデバイスを作成し、
始点をns2-veth0、終点をgw-veth1とする。
ns2-veth0イーサネットデバイスの始点をns2、終点をrouterと紐付ける。
ns2側のns2-veth0イーサネットデバイスのエンドポイントを、198.51.100.1/24、
router側のns2-veth0イーサネットデバイスのエンドポイントを、198.51.100.254/24、
に設置する。
それぞれのエンドポイントを起動する。

ns1のルートテーブルにデフォルトルートとrouter側のns1-veth0イーサネットデバイスのエンドポイントを設定する。
ns2のルートテーブルにデフォルトルートとrouter側のns2-veth0イーサネットデバイスのエンドポイントを設定する。
routerのカーネルにIPv4のルーターとして動作することを設定する。

sudo ip netns add ns1
sudo ip netns add ns2
sudo ip netns add router
sudo ip link add ns1-veth0 type veth peer name gw-veth0
sudo ip link set ns1-veth0 netns ns1
sudo ip link set gw-veth0 netns router

sudo ip link add ns2-veth0 type veth peer name gw-veth1
sudo ip link set ns2-veth0 netns ns2 
sudo ip link set gw-veth1 netns router

sudo ip netns exec ns1 ip link set ns1-veth0 up
sudo ip netns exec ns2 ip link set ns2-veth0 up
sudo ip netns exec router ip link set gw-veth0 up
sudo ip netns exec router ip link set gw-veth1 up

sudo ip netns exec ns1 ip address add 192.0.2.1/24 dev ns1-veth0
sudo ip netns exec router ip address add 192.0.2.254/24 dev gw-veth0

sudo ip netns exec ns2 ip address add 198.51.100.1/24 dev ns2-veth0
sudo ip netns exec router ip address add 198.51.100.254/24 dev gw-veth1

sudo ip netns exec ns1 ping -c 3 192.0.2.254
sudo ip netns exec ns2 ping -c 3 198.51.100.254
sudo ip netns exec ns1 ping -c 3 198.51.100.1

ip route show
sudo ip netns exec ns1 ip route show

sudo ip netns exec ns1 ip route add default via 192.0.2.254 
sudo ip netns exec ns2 ip route add default via 198.51.100.254 

sudo ip netns exec router sysctl net.ipv4.ip_forward=1
